141  float Tet::minAngle()
142  {
143  float min = 180;
144 
145  //each tet has 6 dihedral angles between pairs of faces
146  //compute the face normals for each face
147  vec3 face_normals[4];
148 
149  for (int j=0; j<4; j++) {
150  vec3 v0 = this->verts[(j+1)%4]->pos();
151  vec3 v1 = this->verts[(j+2)%4]->pos();
152  vec3 v2 = this->verts[(j+3)%4]->pos();
153  vec3 normal = normalize(cross(v1-v0,v2-v0));
154 
155  // make sure normal faces 4th (opposite) vertex
156  vec3 v3 = this->verts[(j+0)%4]->pos();
157  vec3 v3_dir = normalize(v3 - v0);
158  if(dot(v3_dir, normal) > 0)
159  normal *= -1;
160 
161  face_normals[j] = normal;
162  }
163  //now compute the 6 dihedral angles between each pair of faces
164  for (int j=0; j<4; j++)
165  for (int k=j+1; k<4; k++) {
166  double dot_product = dot(face_normals[j], face_normals[k]);
167  dot_product = std::min(1.,std::max(dot_product,-1.));
168 
169  double dihedral_angle = 180.0 - acos(dot_product) * 180.0 / PI;
170  dihedral_angle = std::min(180.,std::max(0.,dihedral_angle));
171 
172  if (dihedral_angle < min)
173  min = static_cast<float>(dihedral_angle);
174  }
175  return min;
176  }

286  size_t TetMesh::fixVertexWindup(bool verbose) {
287  if (verbose) std::cout <<
288  "Fixing Vertex wind-up..." << std::endl;
289  size_t count = 0;
290  Status s(tets.size());
291  std::vector<Tet*>::iterator iter = tets.begin();
292  // loop over all tets in the mesh
293  while(iter != tets.end()) {
294  Tet* tet = *iter;
295  vec3 v1 = tet->verts[0]->pos();
296  vec3 v2 = tet->verts[1]->pos();
297  vec3 v3 = tet->verts[2]->pos();
298  vec3 v4 = tet->verts[3]->pos();
299  //check the wind up.
300  if ((v4 - v1).dot((v2 - v1).cross(v3 - v1)) < 0.f) {
301  //re-order since v4 is on the wrong side.
302  Vertex* tmp = tet->verts[2];
303  tet->verts[2] = tet->verts[3];
304  tet->verts[3] = tmp;
305  count++;
306  }
307  iter++;
308  if (verbose)
309  s.printStatus();
310  }
311  if (verbose)
312  s.done();
313 
314  if (verbose) {
315  std::cout << "Fixed " << count << " Tet vertex wind-ups." << std::endl;
316  }
317  return count;
318  }
