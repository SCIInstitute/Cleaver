69 template <typename T>
70 double ScalarField<T>::valueAt(double x, double y, double z) const
71 {
72  x = (x - m_bounds.origin.x)*m_scaleInv.x;
73  y = (y - m_bounds.origin.y)*m_scaleInv.y;
74  z = (z - m_bounds.origin.z)*m_scaleInv.z;
75 
76  if(m_centeringType == CellCentered){
77  x -= 0.5f;
78  y -= 0.5f;
79  z -= 0.5f;
80  }
81 
82  double t = fmod(x,1.0);
83  double u = fmod(y,1.0);
84  double v = fmod(z,1.0);
85 
86  int i0 = (int)floor(x); int i1 = i0+1;
87  int j0 = (int)floor(y); int j1 = j0+1;
88  int k0 = (int)floor(z); int k1 = k0+1;
89 
90 
91  if(m_centeringType == CellCentered)
92  {
93  i0 = clamp(i0, 0, m_w-1);
94  j0 = clamp(j0, 0, m_h-1);
95  k0 = clamp(k0, 0, m_d-1);
96 
97  i1 = clamp(i1, 0, m_w-1);
98  j1 = clamp(j1, 0, m_h-1);
99  k1 = clamp(k1, 0, m_d-1);
100  }
101  else if(m_centeringType == NodeCentered)
102  {
103  i0 = clamp(i0, 0, m_w-2);
104  j0 = clamp(j0, 0, m_h-2);
105  k0 = clamp(k0, 0, m_d-2);
106 
107  i1 = clamp(i1, 0, m_w-2);
108  j1 = clamp(j1, 0, m_h-2);
109  k1 = clamp(k1, 0, m_d-2);
110  }
111 
112 
113  double C000 = m_data[i0 + j0*m_w + k0*m_w*m_h];
114  double C001 = m_data[i0 + j0*m_w + k1*m_w*m_h];
115  double C010 = m_data[i0 + j1*m_w + k0*m_w*m_h];
116  double C011 = m_data[i0 + j1*m_w + k1*m_w*m_h];
117  double C100 = m_data[i1 + j0*m_w + k0*m_w*m_h];
118  double C101 = m_data[i1 + j0*m_w + k1*m_w*m_h];
119  double C110 = m_data[i1 + j1*m_w + k0*m_w*m_h];
120  double C111 = m_data[i1 + j1*m_w + k1*m_w*m_h];
121 
122  return double((1-t)*(1-u)*(1-v)*C000 + (1-t)*(1-u)*(v)*C001 +
123  (1-t)* (u)*(1-v)*C010 + (1-t)* (u)*(v)*C011 +
124  (t)*(1-u)*(1-v)*C100 + (t)*(1-u)*(v)*C101 +
125  (t)* (u)*(1-v)*C110 + (t)* (u)*(v)*C111);
126 }

164 template <typename T>
165 BoundingBox ScalarField<T>::dataBounds() const
166 {
167  switch(m_centeringType){
168  case CellCentered:
169  return BoundingBox(vec3::zero, vec3(m_w, m_h, m_d));
170  break;
171 
172  case NodeCentered:
173  return BoundingBox(vec3::zero, vec3(m_w-1, m_h-1, m_d-1));
174  break;
175 
176  default:
177  std::cerr << "Bad CenterType Defined, Returning empty BoundingBox" << std::endl;
178  return BoundingBox(vec3::zero, vec3::zero);
179  break;
180  }
181 }
