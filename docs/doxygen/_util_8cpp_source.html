102 double pow2(int p) {
103  switch (p) {
104  case -20: return 9.53674e-07;
105  case -19: return 1.90735e-06;
106  case -18: return 3.8147e-06;
107  case -17: return 7.62939e-06;
108  case -16: return 1.52588e-05;
109  case -15: return 3.05176e-05;
110  case -14: return 6.10352e-05;
111  case -13: return 0.0001220703125;
112  case -12: return 0.000244140625;
113  case -11: return 0.00048828125;
114  case -10: return 0.0009765625;
115  case -9: return 0.001953125;
116  case -8: return 0.00390625;
117  case -7: return 0.0078125;
118  case -6: return 0.015625;
119  case -5: return 0.03125;
120  case -4: return 0.0625;
121  case -3: return 0.125;
122  case -2: return 0.25;
123  case -1: return 0.5;
124  case 0: return 1;
125  case 1: return 2;
126  case 2: return 4;
127  case 3: return 8;
128  case 4: return 16;
129  case 5: return 32;
130  case 6: return 64;
131  case 7: return 128;
132  case 8: return 256;
133  case 9: return 512;
134  case 10: return 1024;
135  case 11: return 2048;
136  case 12: return 4096;
137  case 13: return 8192;
138  case 14: return 16384;
139  case 15: return 32768;
140  case 16: return 65536;
141  case 17: return 131072;
142  case 18: return 262144;
143  case 19: return 524288;
144  case 20: return 1048576;
145  default:
146  double ret = 1;
147  if (abs(p) == p)
148  for (int i=0; i<abs(p); i++)
149  ret *= 2.0;
150  else
151  for (int i=0; i<abs(p); i++)
152  ret /= 2.0;
153  return ret;
154  }
