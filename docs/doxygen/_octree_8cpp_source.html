61 void OTCell::subdivide()
62 {
63  if(level == 0)
64  return;
65 
66  for(int i=0; i < 8; i++)
67  {
68  if(!this->children[i])
69  {
70  OTCell *child = new OTCell();
71  child->level = level - 1;
72  child->xLocCode = xLocCode | (((i & _001) >> 0) << child->level);
73  child->yLocCode = yLocCode | (((i & _010) >> 1) << child->level);
74  child->zLocCode = zLocCode | (((i & _100) >> 2) << child->level);
75  child->parent = this;
76  this->children[i] = child;
77 
78  // child bounding box is exactly half the size
79  child->bounds.size = 0.5f*bounds.size;
80 
81  // origin depends on which child
82  child->bounds.origin = bounds.origin + vec3(((i & _001) >> 0)*child->bounds.size.x,
83  ((i & _010) >> 1)*child->bounds.size.y,
84  ((i & _100) >> 2)*child->bounds.size.z);
85  }
86  }

241 OTCell* Octree::addCellAtLevel(int x, int y, int z, unsigned int level)
242 {
243  unsigned int xLocCode = (unsigned int)x; // (x * this->maxVal);
244  unsigned int yLocCode = (unsigned int)y; // (y * this->maxVal);
245  unsigned int zLocCode = (unsigned int)z; // (z * this->maxVal);
246 
247  // figure out where this cell should go
248  OTCell *pCell = this->root();
249  unsigned int nextLevel = this->rootLevel - 1;
250  unsigned int n = nextLevel + 1;
251  unsigned int childBranchBit;
252  unsigned int childIndex;
253 
254  while(n-- && pCell->level > level){
255  childBranchBit = 1 << nextLevel;
256  childIndex = ((xLocCode & childBranchBit) >> (nextLevel))
257  + (((yLocCode & childBranchBit) >> (nextLevel)) << 1)
258  + (((zLocCode & childBranchBit) >> (nextLevel)) << 2);
259  --nextLevel;
260 
261  if(!(pCell->children[childIndex])){
262  OTCell *newCell = new OTCell();
263  newCell->level = pCell->level - 1;
264  newCell->xLocCode = pCell->xLocCode | (childBranchBit & xLocCode);
265  newCell->yLocCode = pCell->yLocCode | (childBranchBit & yLocCode);
266  newCell->zLocCode = pCell->zLocCode | (childBranchBit & zLocCode);
267  newCell->parent = pCell;
268  pCell->children[childIndex] = newCell;
269  }
270 
271  pCell = pCell->children[childIndex];
272  }
273 
274  // return newly created leaf-cell, or existing one
275  return pCell;
276 }
